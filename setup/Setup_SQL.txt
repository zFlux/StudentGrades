CREATE TABLE PUBLIC.student
(id CHAR(10) NOT NULL,
name CHAR(150),
dob DATE,
status CHAR(50),
cpi DOUBLE,
primary key (id));

CREATE TABLE PUBLIC.grade
(studentId CHAR(10) NOT NULL,
year CHAR(10),
gpi DOUBLE,
year_result CHAR(10),
subject1 CHAR(20),
subject2 CHAR(20),
subject3 CHAR(20),
subject4 CHAR(20),
subject5 CHAR(20),
primary key (studentId, year));
